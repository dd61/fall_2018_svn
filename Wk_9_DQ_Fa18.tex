\begin{mdframed}[style=testframe]

	Properties of Logarithms:
	\begin{multicols}{2}
	\begin{enumerate}
		\item $\displaystyle \log_a 1=0$;
		\item $\displaystyle \log_a a=1$;
		\item $\displaystyle \log_a a^x=x$;
		\item $\displaystyle a^{\log_a x}=x$.
	\end{enumerate}
	\end{multicols}
\end{mdframed}
